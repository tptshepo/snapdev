using System.ComponentModel.DataAnnotations;

namespace {{package}}.Extensions
{
  public class NumberNotEmptyAttribute : ValidationAttribute
  {
    public NumberNotEmptyAttribute()
      : base(() => "The {0} field is not valid.")
    {
      MinNumber = 1;
    }

    public int MinNumber { get; set; }

    public override bool IsValid(object value)
    {
      if (value == null)
      {
        return false;
      }

      int modelValue = (int)value;
      if (modelValue < MinNumber)
      {
        return false;
      }

      return true;
    }

  }
}
