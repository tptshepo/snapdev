using AutoMapper;
using System.Linq;
using {{package}}.Controllers.Resources;
using {{package}}.Models;

namespace {{package}}.Mapping
{
  public class MappingProfile : Profile
  {
    public MappingProfile()
    {
      
      CreateMap<{{titlecase}}, KeyValuePairResource>();
      CreateMap<{{titlecase}}QueryResource, {{titlecase}}Query>();
      CreateMap<{{titlecase}}, {{titlecase}}Resource>();
      CreateMap<Save{{titlecase}}Resource, {{titlecase}}>()
        .ForMember(d => d.CreatedOn, opt => opt.Ignore())
        .ForMember(d => d.CreatedBy, opt => opt.Ignore());

    }
  }
}
